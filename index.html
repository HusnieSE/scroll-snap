Snap page when user stops scrolling,
with a customizable configuration
and a consistent cross browser behaviour

Manually bind and unbind
the scrolling element 

const snapObj = new ScrollSnap(container, {
  snapDestinationX: '90%',
  timeout: 100,
  duration: 300,
  threshold: 0.2,
  easing: (t: number) => (
    t < 0.5 ? 2 * t * t : -1 + (4 - 2 * t) * t   
  ),
})

function unbind() {
  console.log('element snapped')
  snapObj.unbind()
}

snapObj.bind(unbind)
